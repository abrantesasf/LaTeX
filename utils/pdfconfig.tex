% Configurações básicas do PDF
\hypersetup{
   unicode=true,
   pdflang={pt-BR},
   bookmarksopen=false,
   bookmarksnumbered=true,
   bookmarksopenlevel=5,
   pdfdisplaydoctitle=true,
   pdfpagemode=UseOutlines,
   pdfstartview=FitH,
   pdfnewwindow=true,
   %pagebackref=true,
   pdfcreator={TeX Live 2025: XeTeX (0.999997) + LaTeX2e (2024-11-01) + TeX (3.141592653) + eTeX (2.6)},
   pdfproducer={xdvipdfmx (20250205)},
   %pdftitle={\@title}, 
   %pdfauthor={\@author},
   %pdfsubject={\imprimirpreambulo},
   %pdfkeywords={\imprimirpalavraschave, \imprimirkeywords}, 
   bookmarksdepth=5,
   citecolor=cyan,      % cor dos links para as referências bibliográficas
   linkcolor=red,       % cor dos links internos (sumário, figuras, rodapé)
   filecolor=magenta,   % dor dos links para arquivos
   urlcolor=blue,       % cor para links URL
   anchorcolor=black    % cor para o texto âncora (alvo)
}

% Desabilita comandos especiais para não ocorrer o warning:
% "Package hyperref Warning: Token not allowed in a PDF string (PDFDocEncoding)"
\pdfstringdefDisableCommands{%
  \def\hfill{}%
  \def\uppercase#1{#1}%
}

% Para a uvvtex2:
\makeatletter
\@ifclassloaded{uvvtex2}%
{%
    \hypersetup{
        pdftitle={\@title}, 
        pdfauthor={\@author},
        pdfsubject={\imprimirpreambulo},
        pdfkeywords={\imprimirpalavraschave, \imprimirkeywords}
    }
}%
{%
    \@ifclassloaded{faesatex2}%
    {%
        \hypersetup{
            pdftitle={\@title}, 
            pdfauthor={\@author},
            pdfsubject={\imprimirpreambulo},
            pdfkeywords={\imprimirpalavraschave, \imprimirkeywords}
        }
    }%
    {%
    }%
}
\makeatother

% Ajusta os links e cores dependendo se o PDF é para divulgação web ou se é
% para impressão final em preto-e-branco. Obs.:
%   colorlinks:   false = links em caixas; true = links coloridos
%   hidelinks:    esconde os links
\makeatletter
\@ifclassloaded{uvvtex2}%
{%
    \ifthenelse{\equal{\imprimirversaoweb}{S}}{% Versão para web
        \hypersetup{colorlinks=true}%
    }{% Versão para impressão
        \hypersetup{hidelinks, colorlinks=false}%
    }
}%
{%
    \@ifclassloaded{faesatex2}%
    {%
        \ifthenelse{\equal{\imprimirversaoweb}{S}}{% Versão para web
            \hypersetup{colorlinks=true}%
        }{% Versão para impressão
            \hypersetup{hidelinks, colorlinks=false}%
        }
    }%
    {%
    }%
}%
\makeatother
